Create procedure addProvincia(_nome_p varchar(50),_nome_prov varchar(50))
Begin 
	declare codp int;
	
	Start transaction; 
	Insert into paises (id,nome) values (default,nome_p);
	Select id into codp from paises where nome=_nome_p;
	
	if(codp is not null) then
		begin
			insert into provincia (id,codpais,nome) values(default,codp,_nome_prov);				
			commit;	
		end;
	else
		Select 'Cadastro invalido' as Mensagem;	
		rollback;
	end if;		
end;
